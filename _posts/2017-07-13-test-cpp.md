#include <iostream>
#include <string>
using namespace std;

class Employe
{
    protected:
        string name;
    public:
        Employe(string _name="Nom") : name(_name) {}
        virtual ~Employe(){};
        virtual string getNomFonction() const
        {
            return name;
        }
        virtual int getSalaireAnnuel() const
        {
            return 0;
        }
};

class Stagiaire : public Employe
{
    private:
        int duree;
    public:
        Stagiaire(string _name="Nom", int _duree=0) : Employe(_name), duree(_duree) {}
        virtual string getNomFonction() const
        {
            return Employe::getNomFonction() + " (stagiaire)";
            //Voir pour la conversion
        }

        virtual ~Stagiaire() {}
};

class Salaries : public Employe
{
    protected:
        int numCompteBancaire;
    public:
        Salaries(string _name="Nom", int _numCompteBancaire=0) : 
        Employe(_name), numCompteBancaire(_numCompteBancaire) {}
        virtual ~Salaries() {}
};

class Operateur : public Salaries
{
    private:
        int salaireMensuel;
    public:
        Operateur(string _name="Nom", int _numCompteBancaire=0, int _salaireMensuel=0) : 
        Salaries(_name, _numCompteBancaire), salaireMensuel(_salaireMensuel) {}
        virtual string getNomFonction() const
        {
            return Employe::getNomFonction() + " (operateur)";
        }
        int getSalaireAnnuel() const
        {
            return 12 * salaireMensuel;
        }

        virtual ~Operateur() {}
};

class Vendeur : public Salaries
{
    private:
        int primeAnnuelle;
    public:
        Vendeur(string _name="Nom", int _numCompteBancaire=0, int _primeAnnuelle=0):
         Salaries(_name, _numCompteBancaire), primeAnnuelle(_primeAnnuelle) {}
        virtual string getNomFonction() const
        {
            return Employe::getNomFonction() + " (vendeur)";
        }
        int getSalaireAnnuel() const
        {
            return primeAnnuelle;
        }
        virtual ~Vendeur() {}
};

class Cell
{
    friend class Entreprise;
    private:
        Employe* data;
        Cell* next;
    public:
        Cell(Employe* _data=nullptr, Cell* _next=nullptr) : data(_data), next(_next) {}
        ~Cell() {}
};

class Entreprise
{
    private:
        string nomEntreprise;
        Cell* head;
    public:
        Entreprise(string _nomEntreprise="Entreprise") : nomEntreprise(_nomEntreprise) {head = nullptr;}
        void ajouterEmploye(Employe* _employe)
        {
            if(head==nullptr)
            {
                head = new Cell(_employe);
            }
            else
            {
                Cell *newHead = new Cell(_employe, head);
                head = newHead;
            }
        }
        void afficherNomFonction() const
        {
            cout << nomEntreprise << endl << endl;
            Cell *temp = head;
            while(temp != nullptr)
            {
                cout << temp->data->getNomFonction() << endl;
                temp = temp->next;
            }
            delete temp;
        }
        int calculerSalaireAnnuel() const
        {
            Cell *temp = head;
            int salaireTot = 0;
            while(temp->next != nullptr)
            {
                salaireTot += temp->data->getSalaireAnnuel();
                temp = temp->next;
            }
            delete temp;
            return salaireTot;
        }
        void licencier()
        {
            while(head->next!=nullptr)
            {
                Cell *temp = head;
                head = head->next;
                delete temp;
            }
            head = nullptr;
        }
        ~Entreprise() {}
};
